\begin{Ej}
  For each of the following five integrals, either determine the correct order of integration or, if not possible, state that it is impossible to do so.
  \begin{enumerate}
    \item $\displaystyle\int_0^{1-z}\int_{0}^{y^2}\dd A$.\vspace{3.1cm}
    \item $\displaystyle\int_{0}^{1}\int_{(y-2)/2}^{(2-y)/2}\dd A$.\vspace{3.1cm}
    \item $\displaystyle\int_{-3}^{3}\int_{0}^{\sqrt{9-x^2}}\int_{0}^{\sqrt{9-x^2-y^2}}z\sqrt{x^2+y^2+z^2}\dd V$.\vspace{3.1cm}
    \item $\displaystyle\int_{0}^{2}\int_{0}^{x}\int_{z^2}^{6-z}\dd V$.\vspace{3.1cm}
    \item $\displaystyle\int_{8}^{9}\int_{0}^{x}\int_{z}^{xy}(xy-yz+x^2)\dd V$.\vspace{3.1cm}
  \end{enumerate}
\end{Ej}
